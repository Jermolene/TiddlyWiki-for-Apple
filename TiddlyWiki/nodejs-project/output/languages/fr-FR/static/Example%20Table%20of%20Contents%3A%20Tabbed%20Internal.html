<$macrocall
	$name="toc-tabbed-internal-nav"
	tag="TableOfContents"
	selectedTiddler="$:/temp/toc/selectedTiddler"
	unselectedText="<p>Choisir un article dans la table des matières. Click the arrow to expand a topic.</p>"
	missingText="<p>Tiddler manquant.</p>"
